9.2 crosstalk

https://github.com/ropensci/plotly/blob/master/demo/crosstalk-highlight-epl.R

df_crosstalk <- crosstalk::SharedData$new(df, ~year)

crosstalk::filter_select(
  id = "year",
  label = "Select a year",
  allLevels = TRUE,
  multiple = FALSE,
  sharedData = df_crosstalk,
  group = ~ factor(year)
)
df_crosstalk %>% 
  plotly::plot_ly(x = ~date,
                  y = ~sales,
                  color = ~city,
                  type = 'scatter', 
                  mode = 'lines')
